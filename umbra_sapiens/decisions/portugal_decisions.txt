				}
			}

			OR = {
				has_landed_title = b_ribatejo-iii
				any_realm_lord = {
					has_landed_title = b_ribatejo-iii
				}
				d_knights_avis = {
					holder_scope = {
						OR = {
							has_landed_title = b_ribatejo-iii
							any_realm_lord = {
								has_landed_title = b_ribatejo-iii
							}
						}
					}
				}
			}

			NOT = {
				has_alternate_start_parameter = { key = religion_names value = random }
			}
		}
		allow = {
			war = no
			wealth = 300
			is_heretic = no
			culture = portuguese
			independent = yes

			conditional_tooltip = {
				trigger = {
					NOT = {
						d_knights_christ = {
							holder_scope = {
								opinion = {
									who = ROOT
									value = 25
								}
							}
						}
					}
				}

				custom_tooltip = {
					text = vassalize_knights_christ_opinion_TT

					d_knights_templar = {
						holder_scope = {
							opinion = {
								who = ROOT
								value = 25
							}
						}
					}
				}
			}
			NOT = {
				trait = excommunicated
			}

			custom_tooltip = {
				text = only_one_vassalized_holy_order_TT

				NOT = {
					any_realm_lord = {
						holy_order = yes
					}
				}
			}
		}
		effect = {
			wealth = -300

			d_knights_christ = {
				show_scope_change = no
				holder_scope = {
					show_scope_change = no
					grant_title_no_opinion = b_ribatejo-iii
				}
			}

			hidden_tooltip = {
				d_knights_christ = {
					save_event_target_as = grandmaster_of_christ
					holder_scope = {
						wealth = 300
					}
					set_defacto_liege = ROOT
				}

				b_ribatejo-iii = {
					any_direct_de_jure_vassal_title = {
						limit = {
							owner = {
								OR = {
									character = ROOT
									any_liege = {
										character = ROOT
									}
								}
							}
						}
						if = {
							limit = {
								owner = {
									character = ROOT
								}
							}
							if = {
								limit = {
									holding_type = CITY
								}
								create_random_steward = {
									random_traits = yes
									dynasty = random
								}
							}
							else_if = {
								limit = {
									holding_type = TEMPLE
								}
								create_random_priest = {
									random_traits = yes
									dynasty = random
								}
							}
							else = {
								create_random_soldier = {
									random_traits = yes
									dynasty = random
								}
							}
							new_character = {
								grant_title_no_opinion = PREV
								set_defacto_liege = event_target:grandmaster_of_christ
							}
						}
						else_if = {
							limit = {
								owner = {
									NOT = {
										character = ROOT
									}
								}
							}
							owner = {
								set_defacto_liege = event_target:grandmaster_of_christ
							}
						}
					}
				}
			}
		}
